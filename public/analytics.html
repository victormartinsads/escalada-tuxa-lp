Como configurar via Google Apps Script? (Clique para ver)
  1. Abra a sua planilha do Google.
  2. Clique no menu superior em Extensões e depois em Apps Script.
  3. Apague todo o código existente e cole o código abaixo.
  4. Clique no ícone de disquete (Salvar) ou pressione Ctrl+S.
  5. Clique no botão azul no canto superior direito: Implantar > Nova implantação.
  6. Clique no ícone de engrenagem ao lado de "Selecionar tipo" e escolha App da Web.
  7. No campo "Quem pode acessar", mude para Qualquer pessoa (crucial para o envio funcionar).
  8. Clique em Implantar. Autorize o acesso se solicitado pelo Google.
  9. Copie o link do "URL" gerado (termina com /exec) e cole no campo acima!
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  try {
    var data = JSON.parse(e.postData.contents);
    if (sheet.getLastRow() === 0) {
      var headers = ["Data", "Nome Completo", "E-mail Profissional", "WhatsApp de Contato"];
      Object.keys(data).forEach(function(k) {
        if (headers.indexOf(k) === -1 && ["id","timestamp","name","email","whatsapp"].indexOf(k) === -1) {
          headers.push(k);
        }
      });
      sheet.appendRow(headers);
    }
    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    var row = [];
    for (var i = 0; i < headers.length; i++) {
      var h = headers[i];
      if (h === "Data") {
        row.push(new Date().toLocaleString("pt-BR"));
      } else if (h === "Nome Completo") {
        row.push(data.name || "");
      } else if (h === "E-mail Profissional") {
        row.push(data.email || "");
      } else if (h === "WhatsApp de Contato") {
        row.push(data.whatsapp || "");
      } else {
        row.push(data[h] || "");
      }
    }
    sheet.appendRow(row);
    return ContentService.createTextOutput(JSON.stringify({status:"success"})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:"error",message:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
